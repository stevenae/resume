                     \vbox{\hsize=\MACdimenA
                           \vskip\boxaroundvmargin #1
                           \vskip\boxaroundvmargin}\hss}%
   \borderandboxit{0pt}\boxrulethickness{\box\MACboxA}}%
  \vskip2pt plus0pt minus0pt
}
